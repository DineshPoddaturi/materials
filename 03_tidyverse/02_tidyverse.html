NY Times model: take monthly means for years 1951 to 1980

require(tidyverse)
ny_model <- temps %>% filter(between(Year, 1951, 1980)) %>% 
  group_by(Month) %>%
  summarise(Temp = mean(Temp, na.rm = TRUE))
